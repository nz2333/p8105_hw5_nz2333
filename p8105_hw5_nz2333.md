fill_nas = function(var) {
  if (is.numeric(var)) {
    var[is.na(var)] = mean(var, na.rm = TRUE)
  }
  else if (!is.numeric(var)) {
    var[is.na(var)] = "virginica"
  }
  return(var)
}
